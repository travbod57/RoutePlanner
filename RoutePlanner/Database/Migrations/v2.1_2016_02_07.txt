  `Id` int(11) NOT NULL AUTO_INCREMENT,
  `TripId` int(11) NOT NULL,
  `Token` char(36) NOT NULL,
  `ExpiryDate` datetime NOT NULL,
  PRIMARY KEY (`Id`),
  KEY `TripId` (`TripId`)
) ENGINE=InnoDB  DEFAULT CHARSET=latin1 AUTO_INCREMENT=3 ;
